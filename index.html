@app.get("/")
async def root():
  return {"message": "Welcome to Haitham's Projects. Use /projects/{project_name} to view a specific project."}

@app.get("/projects/{project_name}")
async def get_project_details(project_name: str):
  project = projects.get(project_name.lower())
  if not project:
    return {"error": "Project not found."}

  return {
    "description": project["description"],
    "tools_used": project["tools"],
    "github_link": project["github"],
    "live_link": project["live_link"]
    }

@app.get("/redirect/github/{project_name}")
async def redirect_github(project_name: str):
  project = projects.get(project_name.lower())
  if not project:
    return {"error": "Project not found."}

  return RedirectResponse(project["github"])

@app.get("/redirect/live/{project_name}")
async def redirect_live(project_name: str):
  project = projects.get(project_name.lower())
  if not project:
    return {"error": "Project not found."}

  return RedirectResponse(project["live_link"])
